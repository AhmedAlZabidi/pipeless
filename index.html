Pipeless

A simple Python library for building a data pipeline.

from pipeless import pipeline

error_handler = lambda item, exception: None
function, run, _ = pipeline(error_handler)


@function
def forwards_and_backwards(_):
    yield _
    yield _[::-1]


@function
def title(_):
    return _.title()

list(run(['tool', 'flow']))
# => ['Tool', 'Loot', 'Flow', 'Wolf']
